13/// Parsed representation of a Soroban contract's WASM definition.
14#[derive(Clone, Debug, Serialize, Deserialize)]
15pub struct ContractDefinition {
16    pub contract_id: String,
17    pub wasm_hash: String,
18    pub network: String,
19    #[serde(skip_serializing_if = "Option::is_none")]
20    pub wasm_bytes: Option<Vec<u8>>,
21    pub cached_at_ms: u128,
22}

24/// Ledger footprint schema describing which keys are read/written during execution.

27    pub contract_id: String,

29    pub read_write: Vec<String>,

33struct CacheEntry<T> {
34    value: T,
35    expires_at: Instant,
36}
37
38impl<T: Clone> CacheEntry<T> {
39    fn new(value: T, ttl: Duration) -> Self {
40        Self {
41            value,
42            expires_at: Instant::now() + ttl,
43        }
44    }
45
46    fn is_expired(&self) -> bool {
47        Instant::now() >= self.expires_at
48    }
49}

51/// Thread-safe TTL cache for Soroban contract definitions and footprint schemas.
52///
53/// Speeds up gasless smart-contract mint validation by avoiding repeated
54/// Horizon/RPC lookups for contracts already seen in the current TTL window.
55pub struct ContractCache {
56    definitions: Arc<Mutex<HashMap<String, CacheEntry<ContractDefinition>>>>,
57    footprints: Arc<Mutex<HashMap<String, CacheEntry<FootprintSchema>>>>,
58    ttl: Duration,
59}
60
61impl ContractCache {
62    pub fn new(ttl_secs: u64) -> Self {
63        Self {
64            definitions: Arc::new(Mutex::new(HashMap::new())),
65            footprints: Arc::new(Mutex::new(HashMap::new())),
66            ttl: Duration::from_secs(ttl_secs),
67        }
68    }
69
70    pub async fn get_definition(&self, contract_id: &str) -> Option<ContractDefinition> {
71        let guard = self.definitions.lock().await;
72        guard
73            .get(contract_id)
74            .filter(|e| !e.is_expired())
75            .map(|e| e.value.clone())
76    }
77
78    pub async fn set_definition(&self, definition: ContractDefinition) {
79        let contract_id = definition.contract_id.clone();
80        let mut guard = self.definitions.lock().await;
81        guard.insert(contract_id.clone(), CacheEntry::new(definition, self.ttl));
82        info!("[ContractCache] Cached definition for contract {}", contract_id);
83    }
84
85    pub async fn get_footprint(&self, contract_id: &str) -> Option<FootprintSchema> {
86        let guard = self.footprints.lock().await;
87        guard
88            .get(contract_id)
89            .filter(|e| !e.is_expired())
90            .map(|e| e.value.clone())
91    }
92
93    pub async fn set_footprint(&self, footprint: FootprintSchema) {
94        let contract_id = footprint.contract_id.clone();
95        let mut guard = self.footprints.lock().await;
96        guard.insert(contract_id.clone(), CacheEntry::new(footprint, self.ttl));
97        info!("[ContractCache] Cached footprint for contract {}", contract_id);
98    }
99
100    pub async fn invalidate(&self, contract_id: &str) {
101        let mut defs = self.definitions.lock().await;
102        let mut foots = self.footprints.lock().await;
103        defs.remove(contract_id);
104        foots.remove(contract_id);
105        info!("[ContractCache] Invalidated cache for contract {}", contract_id);
106    }
107
108    /// Remove all TTL-expired entries. Called periodically by the eviction task.
109    pub async fn evict_expired(&self) {
110        {
111            let mut guard = self.definitions.lock().await;
112            let before = guard.len();
113            guard.retain(|_, entry| !entry.is_expired());
114            let evicted = before - guard.len();
115            if evicted > 0 {
116                info!("[ContractCache] Evicted {} expired definition(s)", evicted);
117            }
118        }
119        {
120            let mut guard = self.footprints.lock().await;
121            let before = guard.len();
122            guard.retain(|_, entry| !entry.is_expired());
123            let evicted = before - guard.len();
124            if evicted > 0 {
125                info!("[ContractCache] Evicted {} expired footprint(s)", evicted);
126            }
127        }
128    }
129
130    pub async fn definition_count(&self) -> usize {
131        self.definitions.lock().await.len()
132    }
133
134    pub async fn footprint_count(&self) -> usize {
135        self.footprints.lock().await.len()
136    }
137}

169    fn sample_definition(id: &str) -> ContractDefinition {
170        ContractDefinition {
171            contract_id: id.to_string(),
172            wasm_hash: "deadbeef1234".to_string(),
173            network: "testnet".to_string(),
174            wasm_bytes: None,
175            cached_at_ms: 0,
176        }
177    }

179    fn sample_footprint(id: &str) -> FootprintSchema {
180        FootprintSchema {
181            contract_id: id.to_string(),

188    #[tokio::test]
189    async fn test_set_and_get_definition() {
190        let cache = ContractCache::new(300);
191        cache.set_definition(sample_definition("CA")).await;
192        let result = cache.get_definition("CA").await.unwrap();

194        assert_eq!(result.wasm_hash, "deadbeef1234");
195    }

202        assert_eq!(result.contract_id, "CB");
203        assert_eq!(result.read_only, vec!["key_ro"]);
204        assert_eq!(result.read_write, vec!["key_rw"]);
205    }
