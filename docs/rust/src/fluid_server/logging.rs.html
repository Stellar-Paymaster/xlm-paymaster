29#[derive(Clone, Copy, Debug, PartialEq, Eq)]
30pub enum LogProvider {
31    Disabled,
32    Datadog,
33    Elk,
34    NewRelic,
35}
36
37impl LogProvider {
38    fn default_endpoint(self) -> Option<&'static str> {
39        match self {
40            Self::Disabled => None,
41            Self::Datadog => Some("https://http-intake.logs.datadoghq.com/api/v2/logs"),
42            Self::Elk => Some("http://localhost:9200/_bulk"),
43            Self::NewRelic => Some("https://log-api.newrelic.com/log/v1"),
44        }
45    }
46
47    fn requires_api_key(self) -> bool {
48        matches!(self, Self::Datadog | Self::NewRelic)
49    }
50}
51
52impl FromStr for LogProvider {
53    type Err = String;
54
55    fn from_str(value: &str) -> Result<Self, Self::Err> {
56        let normalized = value.trim().to_ascii_lowercase();
57        match normalized.as_str() {
58            "" | "none" | "disabled" => Ok(Self::Disabled),
59            "datadog" | "dd" => Ok(Self::Datadog),
60            "elk" | "elasticsearch" => Ok(Self::Elk),
61            "newrelic" | "new_relic" | "nr" => Ok(Self::NewRelic),
62            _ => Err(format!("unsupported log provider: {value}")),
63        }
64    }
65}

94impl LogAggregationConfig {
95    pub fn from_env() -> Result<Self, String> {
96        let provider = std::env::var("FLUID_LOG_AGGREGATION_PROVIDER")
97            .unwrap_or_else(|_| "disabled".to_string())
98            .parse::<LogProvider>()?;
99
100        let endpoint = std::env::var("FLUID_LOG_AGGREGATION_ENDPOINT")
101            .ok()
102            .map(|value| value.trim().to_string())
103            .filter(|value| !value.is_empty())
104            .or_else(|| provider.default_endpoint().map(ToString::to_string));
105
106        let api_key = std::env::var("FLUID_LOG_AGGREGATION_API_KEY")
107            .ok()
108            .map(|value| value.trim().to_string())
109            .filter(|value| !value.is_empty());
110
111        let batch_size = env_parse("FLUID_LOG_AGGREGATION_BATCH_SIZE", DEFAULT_BATCH_SIZE)
112            .clamp(1, 1_000);
113        let flush_ms = env_parse("FLUID_LOG_AGGREGATION_FLUSH_MS", DEFAULT_FLUSH_MS).max(100);
114        let timeout_ms =
115            env_parse("FLUID_LOG_AGGREGATION_TIMEOUT_MS", DEFAULT_TIMEOUT_MS).max(100);
116        let elk_index = std::env::var("FLUID_LOG_AGGREGATION_ELK_INDEX")
117            .ok()
118            .map(|value| value.trim().to_string())
119            .filter(|value| !value.is_empty())
120            .unwrap_or_else(|| "fluid-server-logs".to_string());
121        let service_name = std::env::var("FLUID_SERVICE_NAME")
122            .ok()
123            .map(|value| value.trim().to_string())
124            .filter(|value| !value.is_empty())
125            .unwrap_or_else(|| "fluid-server".to_string());
126
127        let config = Self {
128            provider,
129            endpoint,
130            api_key,
131            batch_size,
132            flush_interval: Duration::from_millis(flush_ms),
133            request_timeout: Duration::from_millis(timeout_ms),
134            elk_index,
135            service_name,
136        };
137        config.validate()?;
138        Ok(config)
139    }
140
141    pub fn validate(&self) -> Result<(), String> {
142        if self.provider == LogProvider::Disabled {
143            return Ok(());
144        }
145
146        let endpoint = self
147            .endpoint
148            .as_deref()
149            .ok_or_else(|| "log aggregation endpoint is required when provider is enabled".to_string())?;
150        if !(endpoint.starts_with("http://") || endpoint.starts_with("https://")) {
151            return Err("log aggregation endpoint must start with http:// or https://".to_string());
152        }
153
154        if self.provider.requires_api_key() && self.api_key.is_none() {
155            return Err("log aggregation API key is required for selected provider".to_string());
156        }
157
158        Ok(())
159    }
160}

209fn init_logging(config: LogAggregationConfig) -> Result<LoggingInitReport, LoggingInitError> {
210    let env_filter = tracing_subscriber::EnvFilter::try_from_default_env()
211        .unwrap_or_else(|_| "fluid_server=info,tower_http=info".into());
212
213    if config.provider == LogProvider::Disabled {
214        tracing_subscriber::registry()
215            .with(env_filter)
216            .with(tracing_subscriber::fmt::layer())
217            .try_init()
218            .map_err(|error| LoggingInitError::Subscriber(error.to_string()))?;
219
220        return Ok(LoggingInitReport {
221            provider: config.provider,
222            endpoint: None,
223        });
224    }
225
226    let (sender, receiver) = mpsc::channel(MAX_QUEUE_SIZE);
227    let endpoint = config.endpoint.clone();
228    tokio::spawn(export_worker(config.clone(), receiver));
229
230    tracing_subscriber::registry()
231        .with(env_filter)
232        .with(tracing_subscriber::fmt::layer().json())
233        .with(AggregationLayer {
234            sender,
235            service_name: config.service_name,
236        })
237        .try_init()
238        .map_err(|error| LoggingInitError::Subscriber(error.to_string()))?;
239
240    Ok(LoggingInitReport {
241        provider: config.provider,
242        endpoint,
243    })
244}

246struct AggregationLayer {
247    sender: mpsc::Sender<AggregatedLog>,
248    service_name: String,
249}
250
251impl<S> Layer<S> for AggregationLayer
252where
253    S: Subscriber + for<'lookup> LookupSpan<'lookup>,
254{
255    fn on_event(&self, event: &tracing::Event<'_>, _ctx: Context<'_, S>) {
256        let mut visitor = FieldVisitor::default();
257        event.record(&mut visitor);
258
259        let record = AggregatedLog {
260            timestamp_ms: now_ms(),
261            level: event.metadata().level().to_string(),
262            target: event.metadata().target().to_string(),
263            fields: visitor.into_json(),
264            service: self.service_name.clone(),
265        };
266
267        if self.sender.try_send(record).is_err() {
268            tracing::warn!("log aggregation queue is full; dropping event");
269        }
270    }
271}

273#[derive(Default)]
274struct FieldVisitor {
275    values: serde_json::Map<String, serde_json::Value>,
276}
277
278impl FieldVisitor {
279    fn insert_str(&mut self, key: &str, value: &str) {
280        let trimmed = if value.len() > MAX_FIELD_VALUE_BYTES {
281            &value[..MAX_FIELD_VALUE_BYTES]
282        } else {
283            value
284        };
285        self.values
286            .insert(key.to_string(), serde_json::Value::String(trimmed.to_string()));
287    }
288
289    fn into_json(self) -> serde_json::Value {
290        serde_json::Value::Object(self.values)
291    }
292}
293
294impl tracing::field::Visit for FieldVisitor {
295    fn record_bool(&mut self, field: &tracing::field::Field, value: bool) {
296        self.values
297            .insert(field.name().to_string(), serde_json::Value::Bool(value));
298    }
299
300    fn record_f64(&mut self, field: &tracing::field::Field, value: f64) {
301        if let Some(number) = serde_json::Number::from_f64(value) {
302            self.values
303                .insert(field.name().to_string(), serde_json::Value::Number(number));
304        }
305    }
306
307    fn record_i64(&mut self, field: &tracing::field::Field, value: i64) {
308        self.values.insert(
309            field.name().to_string(),
310            serde_json::Value::Number(value.into()),
311        );
312    }
313
314    fn record_u64(&mut self, field: &tracing::field::Field, value: u64) {
315        self.values.insert(
316            field.name().to_string(),
317            serde_json::Value::Number(value.into()),
318        );
319    }
320
321    fn record_str(&mut self, field: &tracing::field::Field, value: &str) {
322        self.insert_str(field.name(), value);
323    }
324
325    fn record_error(
326        &mut self,
327        field: &tracing::field::Field,
328        value: &(dyn std::error::Error + 'static),
329    ) {
330        self.insert_str(field.name(), &value.to_string());
331    }
332
333    fn record_debug(&mut self, field: &tracing::field::Field, value: &dyn fmt::Debug) {
334        self.insert_str(field.name(), &format!("{value:?}"));
335    }

345async fn export_worker(config: LogAggregationConfig, mut receiver: mpsc::Receiver<AggregatedLog>) {
346    let mut interval = time::interval(config.flush_interval);
347    interval.set_missed_tick_behavior(time::MissedTickBehavior::Skip);
348
349    let client = reqwest::Client::builder()
350        .timeout(config.request_timeout)
351        .build();
352    let Ok(client) = client else {
353        eprintln!("[log-aggregation] failed to build HTTP client");
354        return;
355    };
356
357    // Consecutive remote-export failures before falling back to disk.
358    let mut consecutive_failures: u32 = 0;
359    const FALLBACK_THRESHOLD: u32 = 3;
360
361    let mut batch = Vec::with_capacity(config.batch_size);
362    loop {
363        tokio::select! {
364            Some(record) = receiver.recv() => {
365                batch.push(record);
366                if batch.len() >= config.batch_size {
367                    let failed = flush_batch(&client, &config, &mut batch).await;
368                    if failed {
369                        consecutive_failures += 1;
370                        if consecutive_failures >= FALLBACK_THRESHOLD {
371                            flush_batch_to_disk(&config, &mut batch);
372                        }
373                    } else {
374                        consecutive_failures = 0;
375                    }
376                }
377            }
378            _ = interval.tick() => {
379                if !batch.is_empty() {
380                    let failed = flush_batch(&client, &config, &mut batch).await;
381                    if failed {
382                        consecutive_failures += 1;
383                        if consecutive_failures >= FALLBACK_THRESHOLD {
384                            flush_batch_to_disk(&config, &mut batch);
385                        }
386                    } else {
387                        consecutive_failures = 0;
388                    }
389                }
390            }
391            else => {
392                if !batch.is_empty() {
393                    let failed = flush_batch(&client, &config, &mut batch).await;
394                    if failed {
395                        flush_batch_to_disk(&config, &mut batch);
396                    }
397                }
398                break;
399            }
400        }
401    }
402}

404async fn flush_batch(client: &reqwest::Client, config: &LogAggregationConfig, batch: &mut Vec<AggregatedLog>) -> bool {
405    let payload = match build_payload(config, batch) {
406        Ok(payload) => payload,
407        Err(error) => {
408            eprintln!("[log-aggregation] failed to build payload: {error}");
409            batch.clear();
410            return true; // treat serialisation failure as a remote failure
411        }
412    };
413
414    let Some(endpoint) = config.endpoint.as_deref() else {
415        batch.clear();
416        return false;
417    };
418
419    let mut request = client.post(endpoint);
420    match config.provider {
421        LogProvider::Datadog => {
422            if let Some(api_key) = config.api_key.as_deref() {
423                request = request.header("DD-API-KEY", api_key);
424            }
425            request = request.header("Content-Type", "application/json");
426        }
427        LogProvider::NewRelic => {
428            if let Some(api_key) = config.api_key.as_deref() {
429                request = request.header("Api-Key", api_key);
430            }
431            request = request.header("Content-Type", "application/json");
432        }
433        LogProvider::Elk => {
434            request = request.header("Content-Type", "application/x-ndjson");
435        }
436        LogProvider::Disabled => {}
437    }
438
439    let started = Instant::now();
440    let failed = match request.body(payload).send().await {
441        Ok(response) if response.status().is_success() => {
442            let elapsed = started.elapsed().as_millis();
443            tracing::debug!("log batch exported in {elapsed}ms");
444            false
445        }
446        Ok(response) => {
447            eprintln!(
448                "[log-aggregation] export failed with status {}",
449                response.status()
450            );
451            true
452        }
453        Err(error) => {
454            eprintln!("[log-aggregation] export request failed: {error}");
455            true
456        }
457    };
458
459    batch.clear();
460    failed
461}

463/// Write a batch of log records to a local fallback file when the remote
464/// aggregation service is unavailable.
465///
466/// The file path is controlled by `FLUID_LOG_FALLBACK_PATH` (default:
467/// `fluid-server-fallback.log` in the current working directory).  Each
468/// record is written as a single JSON line (NDJSON) so the file can be
469/// ingested by any log shipper once the remote service recovers.
470fn flush_batch_to_disk(config: &LogAggregationConfig, batch: &mut Vec<AggregatedLog>) {
471    if batch.is_empty() {
472        return;
473    }
474
475    let path: PathBuf = std::env::var("FLUID_LOG_FALLBACK_PATH")
476        .map(PathBuf::from)
477        .unwrap_or_else(|_| PathBuf::from("fluid-server-fallback.log"));
478
479    let file_result = OpenOptions::new().create(true).append(true).open(&path);
480
481    match file_result {
482        Ok(mut file) => {
483            for record in batch.iter() {
484                match serde_json::to_string(record) {
485                    Ok(line) => {
486                        if let Err(e) = writeln!(file, "{line}") {
487                            eprintln!("[log-aggregation] disk write error: {e}");
488                        }
489                    }
490                    Err(e) => {
491                        eprintln!("[log-aggregation] failed to serialise record for disk: {e}");
492                    }
493                }
494            }
495            eprintln!(
496                "[log-aggregation] remote unavailable — {} record(s) written to {}",
497                batch.len(),
498                path.display()
499            );
500        }
501        Err(e) => {
502            eprintln!(
503                "[log-aggregation] could not open fallback log file {}: {e}",
504                path.display()
505            );
506        }
507    }
508
509    batch.clear();
510}

512fn build_payload(config: &LogAggregationConfig, batch: &[AggregatedLog]) -> Result<String, String> {
513    match config.provider {
514        LogProvider::Datadog | LogProvider::NewRelic => {
515            serde_json::to_string(batch).map_err(|error| error.to_string())
516        }
517        LogProvider::Elk => build_elk_bulk_payload(&config.elk_index, batch),
518        LogProvider::Disabled => Ok(String::new()),
519    }
520}
521
522fn build_elk_bulk_payload(index: &str, batch: &[AggregatedLog]) -> Result<String, String> {
523    let mut payload = String::new();
524    let action_line = serde_json::to_string(&serde_json::json!({
525        "index": {
526            "_index": index,
527        }
528    }))
529    .map_err(|error| error.to_string())?;
530
531    for record in batch {
532        payload.push_str(&action_line);
533        payload.push('\n');
534        let line = serde_json::to_string(record).map_err(|error| error.to_string())?;
535        payload.push_str(&line);
536        payload.push('\n');
537    }
538
539    Ok(payload)
540}
