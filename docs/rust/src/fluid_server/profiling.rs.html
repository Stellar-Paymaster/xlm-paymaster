1//! Memory profiling utilities for zero-copy XDR optimization
2
3use std::alloc::{GlobalAlloc, Layout, System};
4use std::sync::atomic::{AtomicUsize, Ordering};
5
6/// A simple allocator wrapper that tracks allocation counts
7pub struct TrackingAllocator;
8
9static ALLOCATION_COUNT: AtomicUsize = AtomicUsize::new(0);
10static DEALLOCATION_COUNT: AtomicUsize = AtomicUsize::new(0);
11static BYTES_ALLOCATED: AtomicUsize = AtomicUsize::new(0);
12
13unsafe impl GlobalAlloc for TrackingAllocator {
14    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
15        let ptr = System.alloc(layout);
16        if !ptr.is_null() {
17            ALLOCATION_COUNT.fetch_add(1, Ordering::Relaxed);
18            BYTES_ALLOCATED.fetch_add(layout.size(), Ordering::Relaxed);
19        }
20        ptr
21    }
22
23    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
24        System.dealloc(ptr, layout);
25        DEALLOCATION_COUNT.fetch_add(1, Ordering::Relaxed);
26    }
27}

29/// Memory statistics for profiling
30#[derive(Debug, Clone)]
31pub struct MemoryStats {
32    pub allocations: usize,
33    pub deallocations: usize,
34    pub bytes_allocated: usize,
35    pub net_allocations: usize,
36}
37
38impl MemoryStats {
39    pub fn current() -> Self {
40        let allocations = ALLOCATION_COUNT.load(Ordering::Relaxed);
41        let deallocations = DEALLOCATION_COUNT.load(Ordering::Relaxed);
42        let bytes_allocated = BYTES_ALLOCATED.load(Ordering::Relaxed);
43        
44        Self {
45            allocations,
46            deallocations,
47            bytes_allocated,
48            net_allocations: allocations.saturating_sub(deallocations),
49        }
50    }
51
52    pub fn reset() {
53        ALLOCATION_COUNT.store(0, Ordering::Relaxed);
54        DEALLOCATION_COUNT.store(0, Ordering::Relaxed);
55        BYTES_ALLOCATED.store(0, Ordering::Relaxed);
56    }
57
58    pub fn diff(&self, other: &MemoryStats) -> MemoryStats {
59        MemoryStats {
60            allocations: self.allocations.saturating_sub(other.allocations),
61            deallocations: self.deallocations.saturating_sub(other.deallocations),
62            bytes_allocated: self.bytes_allocated.saturating_sub(other.bytes_allocated),
63            net_allocations: self.net_allocations.saturating_sub(other.net_allocations),
64        }
65    }
66}

68/// Benchmark function to measure memory usage of XDR operations
69pub fn benchmark_xdr_parsing<F>(name: &str, iterations: usize, mut operation: F) 
70where 
71    F: FnMut() -> Result<(), Box<dyn std::error::Error>>,
72{
73    // Reset counters
74    MemoryStats::reset();
75    
76    let start_stats = MemoryStats::current();
77    let start_time = std::time::Instant::now();
78    
79    for _ in 0..iterations {
80        if let Err(e) = operation() {
81            eprintln!("Benchmark error in {}: {}", name, e);
82            return;
83        }
84    }
85    
86    let end_time = std::time::Instant::now();
87    let end_stats = MemoryStats::current();
88    let diff_stats = end_stats.diff(&start_stats);
89    
90    println!("\n=== {} Benchmark Results ===", name);
91    println!("Iterations: {}", iterations);
92    println!("Duration: {:?}", end_time - start_time);
93    println!("Total allocations: {}", diff_stats.allocations);
94    println!("Total bytes allocated: {}", diff_stats.bytes_allocated);
95    println!("Allocations per iteration: {:.2}", diff_stats.allocations as f64 / iterations as f64);
96    println!("Bytes per iteration: {:.2}", diff_stats.bytes_allocated as f64 / iterations as f64);
97}
