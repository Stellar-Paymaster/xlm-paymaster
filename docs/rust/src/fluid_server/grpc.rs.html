27#[tonic::async_trait]
28impl SignerService for FluidSignerGrpc {
29    async fn sign(&self, request: Request<SignRequest>) -> Result<Response<SignResponse>, Status> {
30        // Extract client IP or use a default key
31        let client_ip = request
32            .remote_addr()
33            .map(|addr| addr.to_string())
34            .unwrap_or_else(|| "unknown".to_string());
35
36        // Check rate limit (100 requests per 60 seconds per IP)
37        let rate_limit_key = format!("grpc-sign:{}", client_ip);
38        if !self
39            .rate_limiter
40            .is_allowed_with_limits(&rate_limit_key, 100, 60_000)
41            .await
42        {
43            return Err(Status::resource_exhausted(
44                "Rate limit exceeded for this endpoint",
45            ));
46        }
47
48        let request = request.into_inner();
49
50        if request.xdr.trim().is_empty() {
51            return Err(Status::invalid_argument("xdr is required"));
52        }
53        if request.secret_key.trim().is_empty() {
54            return Err(Status::invalid_argument("secret_key is required"));
55        }
56        if request.network_passphrase.trim().is_empty() {
57            return Err(Status::invalid_argument("network_passphrase is required"));
58        }
59
60        info!(
61            "SignRequest received | xdr_len={} | client={}",
62            request.xdr.len(),
63            client_ip
64        );
65
66        let result = sign_transaction_xdr_internal(
67            &request.xdr,
68            &request.secret_key,
69            &request.network_passphrase,
70        )
71        .map_err(map_signing_error)?;
72
73        info!(
74            "SignResponse sent | signer={} signatures={} | client={}",
75            result.signer_public_key, result.signature_count, client_ip
76        );
77
78        Ok(Response::new(SignResponse {
79            signed_xdr: result.signed_xdr,
80            signer_public_key: result.signer_public_key,
81            transaction_hash_hex: result.transaction_hash_hex,
82            signature_count: result.signature_count as u32,
83        }))
84    }
85}
86
87fn map_signing_error(error: SigningError) -> Status {
88    match error {
89        SigningError::InvalidSecretKey(message) => Status::invalid_argument(message),
90        SigningError::InvalidEnvelope(message) => Status::invalid_argument(message),
91        SigningError::UnsupportedEnvelope(message) => Status::failed_precondition(message),
92        SigningError::SignatureOverflow => Status::resource_exhausted(error.to_string()),
93        SigningError::AccountBlocked(message) => Status::permission_denied(message),
94        SigningError::SuspiciousActivity(message) => Status::permission_denied(message),
95    }
96}

98pub async fn serve_grpc(addr: SocketAddr) -> Result<(), Box<dyn std::error::Error>> {
99    info!("XLM Paymaster signer gRPC listening on {addr}");
100
101    // Initialize rate limiter: 100 requests per 60 seconds per IP
102    let rate_limiter = Arc::new(RateLimiter::new(100, 60_000));
103
104    let signer = FluidSignerGrpc::new(rate_limiter.clone());
105
106    Server::builder()
107        .add_service(SignerServiceServer::new(signer))
108        .serve(addr)
109        .await?;
110
111    Ok(())
112}
