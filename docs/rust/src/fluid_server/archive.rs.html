4use aws_config::BehaviorVersion;
5use aws_sdk_s3::Client as S3Client;
6use chrono::{DateTime, Duration, Utc};
7use sqlx::{query, query_as, PgPool};
8use tracing::{error, info};
9
10const BATCH_SIZE: i64 = 1000;
11const OLDER_THAN_DAYS: i64 = 730; // 2 years
12
13#[derive(sqlx::FromRow, serde::Serialize)]
14struct Transaction {
15    id: String,
16    tx_hash: Option<String>,
17    inner_tx_hash: String,
18    tenant_id: Option<String>,
19    status: String,
20    cost_stroops: i64,
21    category: String,
22    chain: String,
23    created_at: DateTime<Utc>,
24}
25
26pub async fn run_archival_job(pool: &PgPool) -> Result<(), Box<dyn std::error::Error>> {
27    info!("Starting transaction archival job");
28
29    let cutoff_date = Utc::now() - Duration::days(OLDER_THAN_DAYS);
30    info!("Archiving transactions older than: {}", cutoff_date);
31
32    // Configure AWS S3
33    let config = aws_config::load_defaults(BehaviorVersion::latest()).await;
34    let s3_client = S3Client::new(&config);
35    let bucket_name =
36        std::env::var("S3_ARCHIVE_BUCKET").expect("S3_ARCHIVE_BUCKET environment variable not set");
37    let prefix = std::env::var("S3_ARCHIVE_PREFIX").unwrap_or_else(|_| "transactions/".to_string());
38
39    let mut total_archived = 0;
40    let mut batch_num = 0;
41
42    loop {
43        let transactions = query_as::<_, Transaction>(
44            r#"
45            SELECT id, "txHash" as tx_hash, "innerTxHash" as inner_tx_hash, 
46                   "tenantId" as tenant_id, status, "costStroops" as cost_stroops,
47                   category, chain, "createdAt" as created_at
48            FROM "Transaction"
49            WHERE "createdAt" < $1
50            ORDER BY "createdAt" ASC
51            LIMIT $2
52            "#,
53        )
54        .bind(cutoff_date)
55        .bind(BATCH_SIZE)
56        .fetch_all(pool)
57        .await?;
58
59        if transactions.is_empty() {
60            break;
61        }
62
63        // Convert to JSON Lines format
64        let json_lines: Vec<String> = transactions
65            .iter()
66            .map(|tx| serde_json::to_string(tx).unwrap())
67            .collect();
68        let body = json_lines.join("\n");
69
70        // Upload to S3
71        let key = format!(
72            "{}{}_{}.jsonl",
73            prefix,
74            Utc::now().format("%Y%m%d_%H%M%S"),
75            batch_num
76        );
77
78        s3_client
79            .put_object()
80            .bucket(&bucket_name)
81            .key(&key)
82            .body(aws_sdk_s3::primitives::ByteStream::from(body.into_bytes()))
83            .content_type("application/x-ndjson")
84            .send()
85            .await?;
86
87        // Delete archived transactions
88        let ids: Vec<String> = transactions.iter().map(|tx| tx.id.clone()).collect();
89        for id in ids {
90            query(r#"DELETE FROM "Transaction" WHERE id = $1"#)
91                .bind(id)
92                .execute(pool)
93                .await?;
94        }
95
96        total_archived += transactions.len();
97        batch_num += 1;
98        info!(
99            "Batch {}: Archived {} transactions to s3://{}/{}",
100            batch_num,
101            transactions.len(),
102            bucket_name,
103            key
104        );
105    }
106
107    info!(
108        "Archival job completed. Total archived: {} transactions",
109        total_archived
110    );
111    Ok(())
112}
